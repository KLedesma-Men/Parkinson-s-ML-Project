#data
import pandas as pd

df = pd.read_csv("/Users/karlyn/Downloads/ReplicatedAcousticFeatures-ParkinsonDatabase.csv")
# Columns that are not acoustic features
id_cols = ["ID", "Recording"]
target_col = "Status"
meta_cols = ["Gender"]

# Feature columns = everything else
feature_cols = [c for c in df.columns if c not in id_cols + [target_col] + meta_cols]

# Group by patient ID and collapse triplicates
df_patient = df.groupby("ID").agg(
    {
        "Status": "max",       
        "Gender": "first",     
        **{c: "mean" for c in feature_cols}  # average all acoustic features
    }
).reset_index()

df_patient.head()
#target vector 
y = df_patient["Status"] # 0=control, 1= Parkinson's 
#Feature matrix
X = df_patient.drop(columns=["Status", "ID"])
